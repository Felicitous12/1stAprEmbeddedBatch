#include<stdio.h>
void main()
{

    char str[35];                            //DECLARATION OF STRING
    printf("Enter the string");
    scanf("%s",str);                         //SCANNING THE DATA IN STRING TYPE
    int size=sizeof(str)/sizeof(str[0]);     //FINDING THE SIZE OF STR
    printf("%d\n",size);
    int i,j,c1=0,c2=0,c3=1;

    /*
    LOGIC FOR
    */

    for(i=0;str[i];i++)
    {
        if(str[i]>='A' && str[i]<='Z')
        {
            c1++;
        }
        else if(str[i]>='0' && str[i]<='9')
        {
            c2++;
        }
    }
    printf("ALPHABETS=%d  NUMBERS=%d\n",c1,c2);


    for(i=0;str[i];i++)
    {

        for(j=i+1,c3=1;str[j];j++)
        {
            if(str[i]==str[j])
            {
                c3++;
            }
        }

     printf("%c is repeated %d times\n",str[i],c3);
    }
}
