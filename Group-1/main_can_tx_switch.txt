#define _XTAL_FREQ 20000000
#include <xc.h>
static void can_init_transceiver(void)
{
    TRISBbits.RB3 = 1;/*configured as CAN_RX (input)*/
    TRISBbits.RB2 = 0;/*configured as CAN_TX (output)*/
}
static void can_init_baudrate(void)
{
    CANCON |= 0x80;/*requests configuration mode*/
    while(CANSTAT != 0x80);/*condition checks for configuration mode*/
    /* Baud rate is 250kHz*/    
    BRGCON1 = 0xC1;/*Sync Jump Width = 4;Baud Rate Pre scaler bits = 4*/
    BRGCON2 = 0xAE;/*Phase Segment 1 Time = 6, Propagation Time = 7*/
    BRGCON3 = 0x45;/*Phase Segment 2 Time = 6*/
    CIOCON = 0x20;  /*CANTX pin will drive VDD when recessive*/
}
static void can_transmit_switch_pressed(void)
{
    CANCON = 0x08;/*requests normal mode and transmit buffer 0*/
    /*Condition checks for Automatically cleared when the message is successfully sent
     requests normal mode and transmit buffer 0*/
    if(TXB0CONbits.TXREQ == 0 && CANCON == 0x08) 
    {
    /*given Standard data frame identifier is 0x570 */
    TXB0SIDH = 0xAE;		/* Standard Identifier higher bits D10 to D3 */
	TXB0SIDL = 0x00;		/* Standard Identifier lower bits D2 TO D0 */
	TXB0DLC = 0x08;			/* Data Length Code    8 bytes of length  */ 
	TXB0D0 = 0x00;			/* DataByte 0  */
	TXB0D1 = 0x00;			/* DataByte 1 */
	TXB0D2 = 0x00;			/* DataByte 2 */
	TXB0D3 = 0x00;			/* DataByte 3 */
    TXB0D4 = 0x00;      /* DataByte 4 */
    TXB0D5 = 0x00;/* DataByte 5 */
    TXB0D6 = 0x00;/* DataByte 6 */
    TXB0D7 = 0x00;/* DataByte 7 */
    TXB0CON = 0x08;         /*message was not aborted,message not lose the arbitration while tx
                            ,bus error did not occur while the message was being sent
                            , Requests sending a message. Clears the TXABT, TXLARB and TXERR bits. 
                                                  , Priority Level 0 (lowest priority                            */
    }
}
static void can_transmit_switch_not_pressed(void)
{
    CANCON = 0x08;/*requests normal mode and transmit buffer 0*/
    /*Condition checks for Automatically cleared when the message is successfully sent
     requests normal mode and transmit buffer 0*/
    if(TXB0CONbits.TXREQ == 0 && CANCON == 0x08) 
    {
    /*given Standard data frame identifier is 0x570 */
    TXB0SIDH = 0xAE;		/* Standard Identifier higher bits D10 to D3 */
	TXB0SIDL = 0x00;		/* Standard Identifier lower bits D2 TO D0 */
	TXB0DLC = 0x08;			/* Data Length Code    8 bytes of length  */ 
	TXB0D3 = 0xff;			/* DataByte 3 */
    TXB0CON = 0x08;         /*message was not aborted,message not lose the arbitration while tx
                            ,bus error did not occur while the message was being sent
                            , Requests sending a message. Clears the TXABT, TXLARB and TXERR bits. 
                                                  , Priority Level 0 (lowest priority                            */
    }
}
 static void press_switch(void)  
{
     TRISA=0X00;/* Port A is configured as output*/
     TRISC=0XFF;/* Port C is configured as input*/
     
     PORTAbits.RA0 = 1; /*Initially LED is in ON state*/
    
     while(1)
     {
         if(PORTCbits.RC0 == 1) /*condition checks if the switch is released*/
     {
        
         can_transmit_switch_not_pressed(); /*function call */
        
         PORTAbits.RA0 = 1; /*Glow the led*/
     }
         else /*condition checks if the switch is pressed*/
     {
         
        can_transmit_switch_pressed();/*function call */
         PORTAbits.RA0 = 0;/*Led in off state*/
     }
     __delay_ms(1000);   /*call the function to generate 1000ms delay*/
     }
}
void main(void) {
    can_init_transceiver();/*function call */
    can_init_baudrate();/*function call */
    press_switch();/*function call */
        
        
   
        
        
    
    
    
}
